let trace1 =
    Scatter3d(
        x = x1,
        y = y1,
        z = z1,
        text = text,
        mode = "markers",
        marker =
            Marker(
                size = 12.,
                line =
                    Line(
                        color = "rgba(217, 217, 217, 0.14)",
                        width = 0.5
                    ),
                opacity = 0.8
            )
    )

let trace2 =
    Scatter3d(
        x = x2,
        y = y2,
        z = z2,
        text = text2,
        mode = "markers",
        marker =
            Marker(
                color = "rgb(127, 127, 127)",
                size = 12.,
                symbol = "circle",
                line =
                    Line(
                        color = "rgb(204, 204, 204)",
                        width = 1.
                    ),
                opacity = 0.9
            )
    )

[trace1; trace2]
|> Chart.Plot
|> Chart.WithWidth 700
|> Chart.WithHeight 500
